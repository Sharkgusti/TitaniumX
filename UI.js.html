"flex" : "none"; } function titulo(texto) { document.title = texto; } function formatoUSD(valor) { return new Intl.NumberFormat("en-US", { style: "currency", currency: "USD" }).format(valor); } function formatoARS(valor) { return new Intl.NumberFormat("es-AR", { style: "currency", currency: "ARS" }).format(valor); } return { toast, loading, titulo, formatoUSD, formatoARS }; })();
